1. Controle de campos

Vamos criar uma tarefa. O usuário digita o nome da tarefa e criamos um novo <input> com <label>: 

<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>JS</title>
    <script>
        window.onload = inicio;

        function inicio() {
            let botao = document.getElementById('btnAdiciona');
            botao.addEventListener('click', adiciona);
        }
        function adiciona() {
            let campoNome = document.getElementById('nome');
            let nomeAtividade = campoNome.value;

            let atividades = document.getElementById('atividades');
            atividades.innerHTML = '<input type="checkbox" id="c1">';
            atividades.innerHTML += '<label>' + nomeAtividade + '</label>';
         }
    </script>
</head>
<body>
    <label for="nome">Atividade:</label>
    <input type="text" id="nome">
    <input type="button" id="btnAdiciona" value="Adicionar"><br>

    <div id="atividades"></div>

</body>
</html>

  1. Ajuste o programa anterior para adicionar mais tarefas.

2. Template String

Podemos inserir valores em Strings usando templates, templates são definidos entre ` e terminam em `: 

let x = 2;

console.log(`Resultado: ${x * 2}`);

As expressões entre ${ e } são avaliadas e o valor é inserido na posição marcada.

Resultando em: 

Resultado: 4

Ao invés de concatenar valores podemos usar templates no código anterior: 

<!DOCTYPE html>

<head>

            botao.addEventListener('click', adiciona);
        }
        function adiciona() {
            let campoNome = document.getElementById('nome');
            let nomeAtividade = campoNome.value;

            let atividades = document.getElementById('atividades');
            atividades.innerHTML = `<input type="checkbox" id="c1">`;
            atividades.innerHTML += `<label> ${nomeAtividade} </label>`;
         }
    </script>
</head>
<body>
    <label for="nome">Atividade:</label>
    <input type="text" id="nome">

3. Valor de checkbox

O campo checkbox possui o atributo checked que pode ser true ou false de acordo com o estado do campo: 

<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>JS</title>
    <script>
        window.onload = inicio;

        function inicio() {
            let botao = document.getElementById('check');
            botao.addEventListener('click', valor);
        }
        function valor() {
            let campo = document.getElementById('check');
            console.log(`Valor: ${campo.value}`);

            console.log(`Apertado: ${campo.checked}`);
         }
    </script>
</head>
<body>
    <input type="checkbox" id="check"><label for="nome">Verificar</label>
</body>
</html>

4. Datas

Em Javascript, a função Date() cria um objeto do tipo Date, para criarmos um objeto com a data atual usamos: 

let agora = new Date();

O objeto possui métodos úteis para obter o dia, mês, ano, hora, minutos atuais e ainda outros dados:

Método Retorno
x.getHours() Hora da data salva em x
x.getMinutes() Minutos
x.getFullYear() Ano
x.getMonth() Mês (janeiro = 0)
x.getDate() Dia do mês
x.getDay() Dia da semana (domingo = 0)

Veja mais na documentação. 

let agora = new Date();

console.log(agora.getHours());

5. Argumento evento

A função executada quando um evento acontece pode receber o evento que gerou: 

<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>JS</title>
    <script>
        window.onload = inicio;

        function inicio() {
            let botao = document.getElementById('check');
            botao.addEventListener('click', valor);
        }

        function valor(evento) {
            let campo = document.getElementById('check');
            console.log(evento);
         }
    </script>
</head>
<body>
    <input type="checkbox" id="check">
    <label for="check">Verificar</label>
</body>
</html>

Dentre os atributos srcElement guarda o objeto que gerou o evento: 

function valor(evento) {
    let campo = document.getElementById('check');
    console.log(evento.srcElement.id);
}

6. EventListener e innerHTML

<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>JS</title>
    <script>
        window.onload = gera;

    function gera() {
        let divChecks = document.getElementById('checks');

        for(var i = 0; i < 5; i++) {
            divChecks.innerHTML += `<input type="checkbox" id="c${i}">`;
            divChecks.innerHTML += `<label for="c${i}">Verificar ${i}</label><br>`;

            let item = document.getElementById(`c${i}`);
            item.addEventListener('click', valor);
        }

    }
        function valor(evento) {
            let campo = document.getElementById('check');
            console.log(evento.srcElement.id);
        }

    </script>
</head>
<body>
    <div id="checks"></div>
</body>
</html>

Usando innerHTML para adicionar itens, somente o último elemento ouve evento click.

Vamos usar createElement para criar os elementos e appendChild para adicionar o elemento ao HTML, assim todos ouvem os seus próprios clicks: 

<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>JS</title>
    <script>
        window.onload = gera;

    function gera() {
        let divChecks = document.getElementById('checks');

        for(var i = 0; i < 5; i++) {
            let checkbox = document.createElement('input');
            checkbox.type = 'checkbox';
            checkbox.id = 'c' + i;

            let label = document.createElement('label');
            label.htmlFor = 'c' + i;
            label.textContent =  `Verificar ${i}`;

            let pula = document.createElement('br');

            divChecks.appendChild(checkbox);
            divChecks.appendChild(label);
            divChecks.appendChild(pula);

            checkbox.addEventListener('click', valor);
        }

    }
        function valor(evento) {
            let campo = document.getElementById('check');
            console.log(evento.srcElement.id);
        }

    </script>
</head>
<body>
    <div id="checks"></div>
</body>
</html>

7. Campo Select

Um campo <select> possui várias opções <option> para o usuário selecionar: 

<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>JS</title>
    <script>
        window.onload = inicio;

        function inicio() {
            let situa = document.getElementById('situacao1');
            // mudar para change
            situa.addEventListener('click', valor);
        }
        function valor() {
            let campo = document.getElementById('situacao1');
            console.log(`Valor: ${campo.value}`);
         }
    </script>
</head>
<body>
    <select id="situacao1">
        <option value="inicio">Aguardando</option>
        <option value="atribuida">Em andamento</option>
        <option value="parada">Pausada</option>
        <option value="rejeitada">Rejeitada</option>
        <option value="fim">Concluída</option>
    </select>
    <label for="situacao1" id="rotulo1">Instalar impressora RH</label>
</body>
</html>
