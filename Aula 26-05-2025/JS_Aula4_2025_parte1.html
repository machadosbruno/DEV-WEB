1. Primeiro JS

<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Hello World!</title>
    <script>
	// página termina de carregar executa inicio()
	window.onload = inicio;

	function inicio() {
	    // elemento com id saida é obtido e armazenado em pSaida
	    let pSaida = document.getElementById('saida');

	    // conteúdo de pSaida é alterado
	    pSaida.innerHTML = 'Hello, world!';
	}
    </script>
</head>
<body>
    <p id="saida"></p>
</body>
</html>

4. value em inteiro ou real

let digitado = "25";
console.log(typeof(digitado));

let numero = parseInt(digitado);
console.log(typeof(numero));

let digitado = "25";
console.log(typeof(digitado));

let numero = parseFloat(digitado);

5. if em Javascript

5.1. if básico

let teste = true;

if( teste ) {
  // executa se teste == true
  console.log('teste verdadeiro!');
}

5.2. if-else

let teste = true;

if( teste ) {
  // executa se teste == true
  console.log('teste verdadeiro!');
} else {
  // executa se teste == false
  console.log('teste falso!');
}

5.3. ternario

Podemos trocar todo o if/else por ?: 

let teste = true;

console.log( teste ? 'teste verdadeiro' : 'teste falso' );

5.4. if/else if/else

let teste1 = true;
let teste2 = true;

if( teste1 ) {
  // executa se teste1 == true
  console.log('teste1 verdadeiro!');
} else if( teste2 ) {
    // executa se teste1 == false e teste2 == true
    console.log('teste2 verdadeiro! teste1 falso');
} else {
  // executa se teste1 == false
    console.log('teste1 falso!');
}

5.5. switch

let cor = 'vermelho';

switch(cor) {
  case 'azul':
    console.log('#0000FF');
    break;
  case 'verde':
    console.log('#00FF00');
    break;
  case 'vermelho':
    console.log('#FF0000');
    break;
  default:
    console.log('#FFFFFF');
    break;
}
